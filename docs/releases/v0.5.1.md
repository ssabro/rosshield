  ```go
  // C-1 Sigstore keyless
  trustedRoot, _ := tuf.LoadDefaultRoot()  // production
  verifier := wasmrt.CosignSigstoreVerifier{
      TrustedMaterial: trustedRoot,
      ExpectedIdentities: []wasmrt.SigstoreIdentity{
          {SubjectRegex: "https://github.com/ssabro/rosshield/.github/workflows/.*",
           Issuer: "https://token.actions.githubusercontent.com"},
      },
  }
  err := verifier.Verify(policyBytes, bundleBytes)

  store := tpm.New(opts)
  ```
